import React from 'react';
import { View, Text, StyleSheet, TouchableOpacity } from 'react-native';
import { Card } from '../../types/Card';
import { useNavigation } from '@react-navigation/native';

interface CardItemProps {
  card: Card;
}

const CardItem: React.FC<CardItemProps> = ({ card }) => {
  const navigation = useNavigation();

  const handleCardPress = () => {
    navigation.navigate('CardDetails', { cardId: card.id });
  };

  return (
    <TouchableOpacity onPress={handleCardPress} style={[styles.container, { backgroundColor: card.color }]}>
      <View>
        <Text style={styles.name}>{card.name}</Text>
        <Text style={styles.number}>{card.number}</Text>
      </View>
        {card.isFavorite && <Text>Preferita</Text>}
        {card.icon && <Text>{card.icon}</Text>}
      <Text>Click: {card.clickCount}</Text>
    </TouchableOpacity>
  );
};

const styles = StyleSheet.create({
  container: {
    padding: 10,
    marginVertical: 5,
    borderRadius: 5,
  },
  name: {
    fontSize: 18,
    fontWeight: 'bold',
  },
  number: {
    fontSize: 14,
  },
});

export default CardItem;
